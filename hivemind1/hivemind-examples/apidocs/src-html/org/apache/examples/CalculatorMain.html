015    package org.apache.examples;
016    
017    import org.apache.hivemind.Registry;
018    
019    /**
020     * Runs through the methods of the {@link org.apache.examples.Calculator} with
021     * a pair of argument values parsed from the command line.
022     *
023     * @author Howard Lewis Ship
024     */
025    public class CalculatorMain
026    {
027        public static void main(String[] args)
028        {
029            double arg0 = Double.parseDouble(args[0]);
030            double arg1 = Double.parseDouble(args[1]);
031    
032            Registry registry = ExampleUtils.buildRegistry("examples.xml");
033    
034            // Since we know there's exactly *one* service-point implementing Calculator,
035            // we can get it this way, and never have to know its service id.
036    
037            Calculator calculator = (Calculator) registry.getService(Calculator.class);
038    
039            System.out.println("Inputs:   " + arg0 + " and " + arg1);
040            System.out.println("Add:      " + calculator.add(arg0, arg1));
041            System.out.println("Subtract: " + calculator.subtract(arg0, arg1));
042            System.out.println("Multiply: " + calculator.multiply(arg0, arg1));
043            System.out.println("Divide:   " + calculator.divide(arg0, arg1));
044            
045            registry.shutdown();
046        }
047    }

015    package org.apache.hivemind;
016    
017    /**
018     * Interface for an object that can create a service's core implementation.
019     * 
020     * @author Howard Lewis Ship
021     */
022    public interface ServiceImplementationFactory
023    {
024        /**
025         * Creates a core implementation object for a particular service extension point. Typically, the
026         * factory creates an instance and modifies it to implement the necessary interface (in much the
027         * same way that an {@link ServiceInterceptorFactory} would). <b>Incompatible change from 1.0: A
028         * long list of individual parameters have been collapsed down into the factoryParameters. </b>
029         * 
030         * @param factoryParameters
031         *            provides the information about the service necessary to construct the service.
032         */
033        public Object createCoreServiceImplementation(
034                ServiceImplementationFactoryParameters factoryParameters);
035    }

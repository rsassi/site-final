015    package org.apache.hivemind;
016    
017    import java.util.List;
018    
019    import org.apache.hivemind.internal.Module;
020    
021    /**
022     * Interface defining an interceptor factory, an object that can create
023     * an interceptor.  Interceptors are objects that implement a particular
024     * interface, adding logic before or after invoking methods on a wrapped
025     * object (implementing the same instance).
026     * 
027     * <p>
028     * Implementations must be stateless and multi-threaded. 
029     * An interceptor may only be applied once to any single service.
030     * The factory will only be invoked once for any single service (even in the case
031     * of non-standard service models such as threaded and pooled).
032     *
033     * @author Howard Lewis Ship
034     */
035    public interface ServiceInterceptorFactory
036    {
037        /**
038          * Creates an interceptor and pushes it onto the interceptor stack.
039          */
040        public void createInterceptor(InterceptorStack stack, Module invokingModule, List parameters);
041    }

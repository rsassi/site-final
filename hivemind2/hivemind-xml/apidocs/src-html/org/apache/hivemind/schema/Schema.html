015    package org.apache.hivemind.schema;
016    
017    import java.util.List;
018    
019    import org.apache.hivemind.Locatable;
020    import org.apache.hivemind.parse.AnnotationHolder;
021    
022    /**
023     * An object which may contain a model, used to identify the form of XML content allowed within some
024     * other, containing element.
025     * <p>
026     * This is very much provisional; in the future will be more control for validation (i.e,
027     * controlling the number of occurances), and support for analogs of W3C SChema sequence and choice.
028     * The excess flexibility here forces some validation into element objects (the objects created from
029     * the {@link org.apache.hivemind.schema.Rule}s within the
030     * {@link org.apache.hivemind.schema.ElementModel}s).
031     * 
032     * @author Howard Lewis Ship
033     */
034    public interface Schema extends AnnotationHolder, Locatable
035    {
036        /**
037         * Returns the (unqualified) id of this schema.
038         * 
039         * @since 1.1
040         */
041        public String getId();
042    
043        /**
044         * Returns a List of {@link ElementModel}, identifing the elements which may be enclosed by the
045         * modeled element.
046         * <p>
047         * The returned list is unmodifiabled and may be empty, but won't be null.
048         */
049        public List getElementModel();
050    
051        /**
052         * Returns true if Elements conforming to this Schema can be indexed using a key. For this to be
053         * true one {@link #getElementModel() top-level ElementModel} must define a non-null
054         * {@link ElementModel#getKeyAttribute() key attribute} or a unique attribute.
055         * 
056         * @since 1.1
057         */
058        public boolean canInstancesBeKeyed();
059    
060        /**
061         * Returns true if the schema is visible to the specified module, false otherwise.
062         * 
063         * @since 1.1
064         */
065        public boolean visibleToModule(String moduleId);
066        
067        /**
068         * 
069         * Return the Module in which the Schema was defined.
070         * 
071         * @since 1.1
072         * 
073         */
074        
075        public String getDefiningModuleId();
076        
077        public String getRootElementClassName();
078        
079        public String getFullyQualifiedId();
080        
081    }

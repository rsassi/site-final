015    package org.apache.hivemind.schema.rules;
016    
017    import java.util.ArrayList;
018    import java.util.List;
019    import java.util.Map;
020    
021    import org.apache.hivemind.HiveMind;
022    import org.apache.hivemind.Location;
023    import org.apache.hivemind.internal.Module;
024    import org.apache.hivemind.schema.Translator;
025    
026    /**
027     * Interprets a string as an extension point id, and provides the elements for that extension point.
028     * Depending on the target property type the elements will be provided as a List or as a Map.
029     * 
030     * @author Howard Lewis Ship
031     */
032    public class ConfigurationTranslator implements Translator
033    {
034        public Object translate(Module contributingModule, Class propertyType, String inputValue,
035                Location location)
036        {
037            if (HiveMind.isBlank(inputValue))
038                return null;
039    
040            // For backward compatibility allow configuratin of type map be assigned to
041            // properties of type List. Drawback: Lazy construction of the configuration
042            // gets lost. The configuration is constructed right here during the conversion
043            Object configuration = contributingModule.getConfiguration(inputValue);
044            if (configuration instanceof Map && propertyType.equals(List.class)) {
045                configuration = new ArrayList(((Map) configuration).values());
046            }
047            
048            return configuration;
049        }
050    }

015    package org.apache.hivemind.definition;
016    
017    import org.apache.hivemind.Orderable;
018    
019    /**
020     * Defines an interceptor for the methods of a {@link ServicePointDefinition service point}.
021     * The interceptor is created by an instance of {@link InterceptorConstructor}.
022     * 
023     * Interceptors are applied in a certain order which bases on the interceptor names. 
024     * If an implementation of this interface wants to effect the order it must
025     * implement the {@link Orderable} interface too.
026     * 
027     * @author Huegen
028     */
029    public interface InterceptorDefinition extends ExtensionDefinition
030    {
031        /**
032         * @return  the constructor for the creation of a interceptor instance.
033         */
034        public InterceptorConstructor getInterceptorConstructor();
035    
036        /**
037         * @return the name of the interceptor. Used for ordering.
038         */
039        public String getName();
040    
041    }

015    package org.apache.examples.annotations.panorama;
016    
017    import java.util.List;
018    
019    import org.apache.examples.panorama.discussions.DiscussionsStartup;
020    import org.apache.examples.panorama.startup.impl.ExecuteStatic;
021    import org.apache.examples.panorama.startup.impl.Task;
022    import org.apache.hivemind.annotations.AbstractAnnotatedModule;
023    import org.apache.hivemind.annotations.definition.Contribution;
024    
025    public class PanoramaDiscussionsModule extends AbstractAnnotatedModule
026    {
027        @Contribution( configurationId="panorama.startup.tasks" )
028        public void contributeTaks(List<Task> tasks)
029        {
030            Task discussionStartupTask = new Task();
031            ExecuteStatic executable = new ExecuteStatic();
032            executable.setTargetClass(DiscussionsStartup.class);
033            discussionStartupTask.setExecutable(executable);
034            discussionStartupTask.setId("discussions");
035            discussionStartupTask.setTitle("Discussions");
036            
037            tasks.add(discussionStartupTask);
038        }
039    }
